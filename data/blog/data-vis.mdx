import pandas as pd
import matplotlib.pyplot as plt
import json
from js import fetch

async def get_json(url):
    resp = await fetch(url)
    txt = await resp.text()
    return json.loads(txt)

# Data Acquisition
data = await get_json(
    "https://api.census.gov/data/2026/cps/basic/feb?get=PEMLR,PWSSWGT,PEMARITL&for=state:*&PEEDUCA=39&key=8a80f2c986224d00771fdaf54351ec4198eddedd"
)

print("Raw data:")
